163  void CSpace::AddEntityToPhysicsEngine(CEmbodiedEntity& c_entity) {
164  /* Get a reference to the root entity */
165  CEntity* pcToAdd = &c_entity.GetRootEntity();
166  /* Get a reference to the position of the entity */
167  const CVector3& cPos = c_entity.GetOriginAnchor().Position;
168  /* Go through engines and check which ones could house the entity */
169  CPhysicsEngine::TVector vecPotentialEngines;
170  for(size_t i = 0; i < m_ptPhysicsEngines->size(); ++i) {
171  if((*m_ptPhysicsEngines)[i]->IsPointContained(cPos)) {
172  vecPotentialEngines.push_back((*m_ptPhysicsEngines)[i]);
173  }
174  }
175  /* If no engine can house the entity, bomb out */
176  if(vecPotentialEngines.empty()) {
177  THROW_ARGOSEXCEPTION("No physics engines available to house entity \"" << pcToAdd->GetId() << "\"@(" << cPos << ").");
178  }
179  /* If the entity is not movable, add the entity to all the matching engines */
180  if(! c_entity.IsMovable()) {
181  bool bAdded = false;
182  for(size_t i = 0; i < vecPotentialEngines.size(); ++i) {
183  bAdded |= vecPotentialEngines[i]->AddEntity(*pcToAdd);
184  }
185  if(!bAdded) {
186  std::ostringstream ossMsg;
187  ossMsg << "None of the matching physics engines (";
188  ossMsg << "\"" << vecPotentialEngines[0]->GetId() << "\"";
189  for(size_t i = 1; i < vecPotentialEngines.size(); ++i) {
190  ossMsg << ",\"" << vecPotentialEngines[i]->GetId() << "\"";
191  }
192  ossMsg << ") can house non-movable entity \"" << pcToAdd->GetId() << "\"@(" << cPos << ").";
193  THROW_ARGOSEXCEPTION(ossMsg.str());
194  }
195  }
196  /* If the entity is movable, only one engine can be associated to the embodied entity */
197  else if(vecPotentialEngines.size() == 1) {
198  /* Only one engine matches, bingo! */
199  if(!vecPotentialEngines[0]->AddEntity(*pcToAdd)) {
200  THROW_ARGOSEXCEPTION("The matching physics engine (\"" << vecPotentialEngines[0]->GetId() << "\"), cannot house movable entity \"" << pcToAdd->GetId() << "\"@(" << cPos << ").");
201  }
202  }
203  else {
204  /* More than one engine matches, pick the first that can manage the entity */
205  for(size_t i = 0; i < vecPotentialEngines.size(); ++i) {
206  if(vecPotentialEngines[i]->AddEntity(*pcToAdd)) return;
207  }
208  /* No engine can house the entity */
209  std::ostringstream ossMsg;
210  ossMsg << "None of the matching physics engines (";
211  ossMsg << "\"" << vecPotentialEngines[0]->GetId() << "\"";
212  for(size_t i = 1; i < vecPotentialEngines.size(); ++i) {
213  ossMsg << ",\"" << vecPotentialEngines[i]->GetId() << "\"";
214  }
215  ossMsg << ") can house movable entity \"" << pcToAdd->GetId() << "\"@(" << cPos << ").";
216  THROW_ARGOSEXCEPTION(ossMsg.str());
217  }
218  }

329  RealNumberGenerator* CreateGenerator(TConfigurationNode& t_tree) {
330  std::string strMethod;
331  GetNodeAttribute(t_tree, "method", strMethod);
332  if(strMethod == "uniform") {
333  CVector3 cMin, cMax;
334  GetNodeAttribute(t_tree, "min", cMin);
335  GetNodeAttribute(t_tree, "max", cMax);
336  if(! (cMin <= cMax)) {
337  THROW_ARGOSEXCEPTION("Uniform generator: the min is not less than or equal to max: " << cMin << " / " << cMax);
338  }
339  return new UniformGenerator(cMin, cMax);
340  }
341  else if(strMethod == "gaussian") {
342  CVector3 cMean, cStdDev;
343  GetNodeAttribute(t_tree, "mean", cMean);
344  GetNodeAttribute(t_tree, "std_dev", cStdDev);
345  return new GaussianGenerator(cMean, cStdDev);
346  }
347  else if(strMethod == "constant") {
348  CVector3 cValues;
349  GetNodeAttribute(t_tree, "values", cValues);
350  return new ConstantGenerator(cValues);
351  }
352  else if(strMethod == "grid") {
353  CVector3 cCenter,cDistances;
354  GetNodeAttribute(t_tree, "center", cCenter);
355  GetNodeAttribute(t_tree, "distances", cDistances);
356  UInt32 unLayout[3];
357  std::string strLayout;
358  GetNodeAttribute(t_tree, "layout", strLayout);
359  ParseValues<UInt32> (strLayout, 3, unLayout, ',');
360  return new GridGenerator(cCenter, unLayout, cDistances);
361  }
362  else {
363  THROW_ARGOSEXCEPTION("Unknown distribution method \"" << strMethod << "\"");
364  }
365  }

410  void CSpace::Distribute(TConfigurationNode& t_tree) {
411  try {
412  /* Get the needed nodes */
413  TConfigurationNode cPositionNode;
414  cPositionNode = GetNode(t_tree, "position");
415  TConfigurationNode cOrientationNode;
416  cOrientationNode = GetNode(t_tree, "orientation");
417  TConfigurationNode cEntityNode;
418  cEntityNode = GetNode(t_tree, "entity");
419  /* Create the real number generators */
420  RealNumberGenerator* pcPositionGenerator = CreateGenerator(cPositionNode);
421  RealNumberGenerator* pcOrientationGenerator = CreateGenerator(cOrientationNode);
422  /* How many entities? */
423  UInt32 unQuantity;
424  GetNodeAttribute(cEntityNode, "quantity", unQuantity);
425  /* How many trials before failing? */
426  UInt32 unMaxTrials;
427  GetNodeAttribute(cEntityNode, "max_trials", unMaxTrials);
428  /* Get the (optional) entity base numbering */
429  UInt64 unBaseNum = 0;
430  GetNodeAttributeOrDefault(cEntityNode, "base_num", unBaseNum, unBaseNum);
431  /* Get the entity type to add (take only the first, ignore additional if any) */
432  TConfigurationNodeIterator itEntity;
433  itEntity = itEntity.begin(&cEntityNode);
434  if(itEntity == itEntity.end()) {
435  THROW_ARGOSEXCEPTION("No entity to distribute specified.");
436  }
437  /* Get the entity base ID */
438  std::string strBaseId;
439  GetNodeAttribute(*itEntity, "id", strBaseId);
440  /* Add the requested entities */
441  for(UInt32 i = 0; i < unQuantity; ++i) {
442  /* Copy the entity XML tree */
443  TConfigurationNode& tEntityTree = *itEntity;
444  /* Set progressive ID */
445  SetNodeAttribute(tEntityTree, "id", strBaseId + ToString(i+unBaseNum));
446  /* Go on until the entity is placed with no collisions or
447  the max number of trials has been exceeded */
448  UInt32 unTrials = 0;
449  bool bDone = false;
450  bool bRetry = false;
451  CEntity* pcEntity;
452  do {
453  /* Create entity */
454  pcEntity = CFactory<CEntity>::New(tEntityTree.Value());
455  /*
456  * Now that you have the entity, check whether the
457  * entity is positional or embodied or has one such
458  * component.
459  *
460  * In case the entity is positional but not embodied,
461  * there's no need to check for collisions.
462  *
463  * In case the entity is embodied, we must check for
464  * collisions.
465  *
466  * To check for collisions, we add the entity in the
467  * place where it's supposed to be, then we ask the
468  * engine if that entity is colliding with something.
469  *
470  * In case of collision, we remove the entity and try a
471  * different position/orientation.
472  */
473  /* Check whether the entity is positional */
474  CPositionalEntity* pcPositionalEntity = GetPositionalEntity(pcEntity);
475  if(pcPositionalEntity != NULL) {
476  /* Set the position */
477  SetNodeAttribute(tEntityTree, "position", (*pcPositionGenerator)(bRetry));
478  /* Set the orientation */
479  SetNodeAttribute(tEntityTree, "orientation", (*pcOrientationGenerator)(bRetry));
480  /* Init the entity (this also creates the components, if pcEntity is a composable) */
481  pcEntity->Init(tEntityTree);
482  /* Wherever we want to put the entity, it's OK, add it */
483  CallEntityOperation<CSpaceOperationAddEntity, CSpace, void>(*this, *pcEntity);
484  bDone = true;
485  }
486  else {
487  /* Assume the entity is embodied */
488  /* If the tree does not have a 'body' node, create a new one */
489  if(!NodeExists(tEntityTree, "body")) {
490  TConfigurationNode tBodyNode("body");
491  AddChildNode(tEntityTree, tBodyNode);
492  }
493  /* Get 'body' node */
494  TConfigurationNode& tBodyNode = GetNode(tEntityTree, "body");
495  /* Set the position */
496  SetNodeAttribute(tBodyNode, "position", (*pcPositionGenerator)(bRetry));
497  /* Set the orientation */
498  SetNodeAttribute(tBodyNode, "orientation", (*pcOrientationGenerator)(bRetry));
499  /* Init the entity (this also creates the components, if pcEntity is a composable) */
500  pcEntity->Init(tEntityTree);
501  /* Check whether the entity is indeed embodied */
502  CEmbodiedEntity* pcEmbodiedEntity = GetEmbodiedEntity(pcEntity);
503  if(pcEmbodiedEntity != NULL) {
504  /* Yes, the entity is embodied */
505  /* Add it to the space and to the designated physics engine */
506  CallEntityOperation<CSpaceOperationAddEntity, CSpace, void>(*this, *pcEntity);
507  /* Check if it's colliding with anything else */
508  if(pcEmbodiedEntity->IsCollidingWithSomething()) {
509  /* Set retry to true */
510  bRetry = true;
511  /* Get rid of the entity */
512  CallEntityOperation<CSpaceOperationRemoveEntity, CSpace, void>(*this, *pcEntity);
513  /* Increase the trial count */
514  ++unTrials;
515  /* Too many trials? */
516  if(unTrials > unMaxTrials) {
517  /* Yes, bomb out */
518  THROW_ARGOSEXCEPTION("Exceeded max trials when trying to distribute objects of type " <<
519  tEntityTree.Value() << " with base id \"" <<
520  strBaseId << "\". I managed to place only " << i << " objects.");
521  }
522  /* Retry with a new position */
523  }
524  else {
525  /* No collision, we're done with this entity */
526  bDone = true;
527  }
528  }
529  else {
530  THROW_ARGOSEXCEPTION("Cannot distribute entities that are not positional nor embodied, and \"" << tEntityTree.Value() << "\" is neither.");
531  }
532  }
533  }
534  while(!bDone);
535  }
536  /* Delete the generators, now unneeded */
537  delete pcPositionGenerator;
538  delete pcOrientationGenerator;
539  }
540  catch(CARGoSException& ex) {
541  THROW_ARGOSEXCEPTION_NESTED("Error while trying to distribute entities", ex);
542  }
543  }
