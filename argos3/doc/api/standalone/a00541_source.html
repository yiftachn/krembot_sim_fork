149  void CLuaController::CreateLuaState() {
150  /* Register functions */
151  CLuaUtility::RegisterLoggerWrapper(m_ptLuaState);
152  /* Register metatables */
153  CLuaVector2::RegisterMetatable(m_ptLuaState);
154  CLuaVector3::RegisterMetatable(m_ptLuaState);
155  CLuaQuaternion::RegisterMetatable(m_ptLuaState);
156  /* Create a table that will contain the state of the robot */
157  lua_newtable(m_ptLuaState);
158  /* Set the id of the robot */
159  lua_pushstring(m_ptLuaState, "id");

161  lua_settable(m_ptLuaState, -3);
162  /* Register RNG */
163  CLuaUtility::RegisterRNG(m_ptLuaState, m_pcRNG);
164  /* Go through devices and add the necessary items to the table */

174  }
175  /* Set the name of the table */
176  lua_setglobal(m_ptLuaState, "robot");
