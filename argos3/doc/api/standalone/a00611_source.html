108  void CFootBotLightRotZOnlySensor::Update() {
109  /* Erase readings */
110  for(size_t i = 0; i < m_tReadings.size(); ++i) {
111  m_tReadings[i].Value = 0.0f;
112  }
113  /* Get foot-bot orientation */
114  CRadians cTmp1, cTmp2, cOrientationZ;
115  m_pcEmbodiedEntity->GetOriginAnchor().Orientation.ToEulerAngles(cOrientationZ, cTmp1, cTmp2);
116  /* Ray used for scanning the environment for obstacles */
117  CRay3 cOcclusionCheckRay;
118  cOcclusionCheckRay.SetStart(m_pcEmbodiedEntity->GetOriginAnchor().Position);
119  CVector3 cRobotToLight;
120  /* Buffer for the angle of the light wrt to the foot-bot */
121  CRadians cAngleLightWrtFootbot;
122  /* Buffers to contain data about the intersection */
123  SEmbodiedEntityIntersectionItem sIntersection;
124  /* List of light entities */
125  CSpace::TMapPerTypePerId::iterator itLights = m_cSpace.GetEntityMapPerTypePerId().find("led"/*"light"*/);
126  if (itLights != m_cSpace.GetEntityMapPerTypePerId().end()) {
127  CSpace::TMapPerType& mapLights = itLights->second;
128  /*
129  * 1. go through the list of light entities in the scene
130  * 2. check if a light is occluded
131  * 3. if it isn't, distribute the reading across the sensors
132  * NOTE: the readings are additive
133  * 4. go through the sensors and clamp their values
134  */
135  for(CSpace::TMapPerType::iterator it = mapLights.begin();
136  it != mapLights.end();
137  ++it) {
138  /* Get a reference to the light */
139  /*CLightEntity*/ CLEDEntity& cLight = *(any_cast</*CLightEntity*/ CLEDEntity *>(it->second));
140  //fprintf(stderr, "id: %s", cLight.GetParent().GetParent().GetId().c_str());
141  if (cLight.GetParent().GetParent().GetId() == m_cRobot->GetId()) { //elhay: if light is coming from our robot, disregard
142  continue;
143  }
144  /* Consider the light only if it has non zero intensity */
145  if(cLight.GetIntensity() > 0.0f) {
146  /* Set the ray end */
147  cOcclusionCheckRay.SetEnd(cLight.GetPosition());
148  /* Check occlusion between the foot-bot and the light */
149  if(! GetClosestEmbodiedEntityIntersectedByRay(sIntersection,
150  cOcclusionCheckRay,
151  *m_pcEmbodiedEntity)) {
152  /* The light is not occluded */
153  if(m_bShowRays) {
154  m_pcControllableEntity->AddCheckedRay(false, cOcclusionCheckRay);
155  }
156  /* Get the distance between the light and the foot-bot */
157  cOcclusionCheckRay.ToVector(cRobotToLight);
158  /*
159  * Linearly scale the distance with the light intensity
160  * The greater the intensity, the smaller the distance
161  */
162  cRobotToLight /= cLight.GetIntensity();
163  /* Get the angle wrt to foot-bot rotation */
164  cAngleLightWrtFootbot = cRobotToLight.GetZAngle();
165  cAngleLightWrtFootbot -= cOrientationZ;
166  /*
167  * Find closest sensor index to point at which ray hits footbot body
168  * Rotate whole body by half a sensor spacing (corresponding to placement of first sensor)
169  * Division says how many sensor spacings there are between first sensor and point at which ray hits footbot body
170  * Increase magnitude of result of division to ensure correct rounding
171  */
172  Real fIdx = (cAngleLightWrtFootbot /*- SENSOR_HALF_SPACING*/) / SENSOR_SPACING;
173  SInt32 nReadingIdx = (fIdx > 0) ? fIdx + 0.5f : fIdx - 0.5f;
174 // fprintf(stderr, "fidx: %f\n", fIdx);
175 // fprintf(stderr, "nReadingIdx: %d\n", nReadingIdx);
176 
177 // if (nReadingIdx == -1) { //elhay: fix for case where light is straigh f/w
178 // nReadingIdx = 0;
179 // }
180  /* Set the actual readings */
181  Real fReading = cRobotToLight.Length();
182 // fprintf(stderr, "fReading: %f\n", fReading);
183  /*
184  * Take 6 readings before closest sensor and 6 readings after - thus we
185  * process sensors that are with 180 degrees of intersection of light
186  * ray with robot body
187  */
188  for(SInt32 nIndexOffset = -1; nIndexOffset < 2; ++nIndexOffset) {
189  UInt32 unIdx = Modulo(nReadingIdx + nIndexOffset, 8);
190 // fprintf(stderr, "nIndexOffset: %d\n", nIndexOffset);
191 // fprintf(stderr, "unIdx: %d\n", unIdx);
192 // fprintf(stderr, "cAngleLightWrtFootbot: %f\n", ToDegrees(cAngleLightWrtFootbot));
193 // fprintf(stderr, "m_tReadings[unIdx].Angle: %f\n", ToDegrees(m_tReadings[unIdx].Angle));
194  CRadians cAngularDistanceFromOptimalLightReceptionPoint = Abs((cAngleLightWrtFootbot - m_tReadings[unIdx].Angle).SignedNormalize());
195 
196 
197  /*
198  * ComputeReading gives value as if sensor was perfectly in line with
199  * light ray. We then linearly decrease actual reading from 1 (dist
200  * 0) to 0 (dist PI/2)
201  */
202 
203 // fprintf(stderr, "cAngularDistanceFromOptimalLightReceptionPoint: %f\n", cAngularDistanceFromOptimalLightReceptionPoint.GetValue());
204 // fprintf(stderr, "ComputeReading(fReading): %f\n", ComputeReading(fReading));
205 // fprintf(stderr, "ScaleReading(cAngularDistanceFromOptimalLightReceptionPoint: %f\n", ScaleReading(cAngularDistanceFromOptimalLightReceptionPoint));
206 
207  m_tReadings[unIdx].Value += ComputeReading(fReading) * ScaleReading(cAngularDistanceFromOptimalLightReceptionPoint);
208 
209 // fprintf(stderr, "Value: %f\n----\n", m_tReadings[unIdx].Value);
210  }
211 
212  }
213  else {
214  /* The ray is occluded */
215  if(m_bShowRays) {
216  m_pcControllableEntity->AddCheckedRay(true, cOcclusionCheckRay);
217  m_pcControllableEntity->AddIntersectionPoint(cOcclusionCheckRay, sIntersection.TOnRay);
218  }
219  }
220  }
221  }
222  /* Apply noise to the sensors */
223  if(m_bAddNoise) {
224  for(size_t i = 0; i < 24; ++i) {
225  m_tReadings[i].Value += m_pcRNG->Uniform(m_cNoiseRange);
226  }
227  }
228  /* Trunc the reading between 0 and 1 */
229  for(size_t i = 0; i < 24; ++i) {
230  SENSOR_RANGE.TruncValue(m_tReadings[i].Value);
231  }
232  }
233  else {
234  /* There are no lights in the environment */
235  if(m_bAddNoise) {
236  /* Go through the sensors */
237  for(UInt32 i = 0; i < m_tReadings.size(); ++i) {
238  /* Apply noise to the sensor */
239  m_tReadings[i].Value += m_pcRNG->Uniform(m_cNoiseRange);
240  /* Trunc the reading between 0 and 1 */
241  SENSOR_RANGE.TruncValue(m_tReadings[i].Value);
242  }
243  }
244  }
245  }
