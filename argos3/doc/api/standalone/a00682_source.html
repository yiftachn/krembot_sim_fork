66  void CBatteryDefaultSensor::Update() {
67  /* Save old charge value (used later for time left estimation) */
68  Real fOldCharge = m_sReading.AvailableCharge;
69  /* Update available charge as seen by the robot */
70  m_sReading.AvailableCharge =
71  m_pcBatteryEntity->GetAvailableCharge() /
72  m_pcBatteryEntity->GetFullCharge();
73  /* Add noise */
74  if(m_bAddNoise) {
75  m_sReading.AvailableCharge += m_pcRNG->Uniform(m_cNoiseRange);
76  /* To trunc battery level between 0 and 1 */
77  UNIT.TruncValue(m_sReading.AvailableCharge);
78  }
79  /* Update time left */
80  Real fDiff = fOldCharge - m_sReading.AvailableCharge;
81  if(Abs(fDiff) > 1e-6) {
82  m_sReading.TimeLeft =
83  fOldCharge *
84  CPhysicsEngine::GetSimulationClockTick() /
85  fDiff;
86  }
87  else {
88  m_sReading.TimeLeft = CPhysicsEngine::GetSimulationClockTick();//std::numeric_limits<Real>::infinity();
89  }
90  }
