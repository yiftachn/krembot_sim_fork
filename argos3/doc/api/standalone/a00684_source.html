127  void CCameraDefaultSensor::Update() {
128  /* vector of controller rays */
129  std::vector<std::pair<bool, CRay3> >& vecCheckedRays =
130  m_pcControllableEntity->GetCheckedRays();
131  /* sensor parameters */
132  CTransformationMatrix3 cWorldToAnchorTransform;
133  CTransformationMatrix3 cWorldToCameraTransform;
134  CTransformationMatrix3 cCameraToWorldTransform;
135  CVector3 cCameraLocation, cLookAt, cUp;
136  CVector3 cX, cY, cZ;
137  CVector3 cNearCenter, cNearTopLeft, cNearTopRight, cNearBottomLeft, cNearBottomRight;
138  CVector3 cFarCenter, cFarTopLeft, cFarTopRight, cFarBottomLeft, cFarBottomRight;
139  std::array<CPlane, 6> arrFrustumPlanes;
140  CVector3 cBoundingBoxMinCorner, cBoundingBoxMaxCorner;
141  CVector3 cBoundingBoxPosition, cBoundingBoxHalfExtents;
142  /* for each camera sensor */
143  for(SSensor& s_sensor : m_vecSensors) {
144  /* calculate transform matrices */
145  cWorldToAnchorTransform.SetFromComponents(s_sensor.Anchor.Orientation, s_sensor.Anchor.Position);
146  cWorldToCameraTransform = cWorldToAnchorTransform * s_sensor.Offset;
147  cCameraToWorldTransform = cWorldToCameraTransform.GetInverse();
148  /* calculate camera direction vectors */
149  cCameraLocation = cWorldToCameraTransform.GetTranslationVector();
150  cLookAt = cWorldToCameraTransform * CVector3::Z;
151  cUp = CVector3(0,-1,0); // -Y
152  cUp.Rotate(cWorldToCameraTransform.GetRotationMatrix());
153  /* calculate direction vectors */
154  cZ = cCameraLocation - cLookAt;
155  cZ.Normalize();
156  cX = cUp;
157  cX.CrossProduct(cZ);
158  cX.Normalize();
159  cY = cZ;
160  cY.CrossProduct(cX);
161  /* calculate frustum coordinates */
162  cNearCenter = cCameraLocation - cZ * s_sensor.Range.GetMin();
163  cFarCenter = cCameraLocation - cZ * s_sensor.Range.GetMax();
164  cNearTopLeft = cNearCenter + (cY * s_sensor.NearPlaneHeight) - (cX * s_sensor.NearPlaneWidth);
165  cNearTopRight = cNearCenter + (cY * s_sensor.NearPlaneHeight) + (cX * s_sensor.NearPlaneWidth);
166  cNearBottomLeft = cNearCenter - (cY * s_sensor.NearPlaneHeight) - (cX * s_sensor.NearPlaneWidth);
167  cNearBottomRight = cNearCenter - (cY * s_sensor.NearPlaneHeight) + (cX * s_sensor.NearPlaneWidth);
168  cFarTopLeft = cFarCenter + (cY * s_sensor.FarPlaneHeight) - (cX * s_sensor.FarPlaneWidth);
169  cFarTopRight = cFarCenter + (cY * s_sensor.FarPlaneHeight) + (cX * s_sensor.FarPlaneWidth);
170  cFarBottomLeft = cFarCenter - (cY * s_sensor.FarPlaneHeight) - (cX * s_sensor.FarPlaneWidth);
171  cFarBottomRight = cFarCenter - (cY * s_sensor.FarPlaneHeight) + (cX * s_sensor.FarPlaneWidth);
172  /* show frustum if enabled by adding outline to the checked rays vector */
173  if(m_bShowFrustum) {
174  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cNearTopRight));
175  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cNearBottomRight));
176  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cNearBottomLeft));
177  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cNearTopLeft));
178  vecCheckedRays.emplace_back(false, CRay3(cFarTopLeft, cFarTopRight));
179  vecCheckedRays.emplace_back(false, CRay3(cFarTopRight, cFarBottomRight));
180  vecCheckedRays.emplace_back(false, CRay3(cFarBottomRight, cFarBottomLeft));
181  vecCheckedRays.emplace_back(false, CRay3(cFarBottomLeft, cFarTopLeft));
182  vecCheckedRays.emplace_back(false, CRay3(cNearTopLeft, cFarTopLeft));
183  vecCheckedRays.emplace_back(false, CRay3(cNearTopRight, cFarTopRight));
184  vecCheckedRays.emplace_back(false, CRay3(cNearBottomRight, cFarBottomRight));
185  vecCheckedRays.emplace_back(false, CRay3(cNearBottomLeft, cFarBottomLeft));
186  }
187  //std::cerr << cFarBottomRight.GetZ() << "\t" << cFarBottomLeft.GetZ() << std::endl; TODO
188  /* generate a bounding box for the frustum */
189  cBoundingBoxMinCorner = cNearCenter;
190  cBoundingBoxMaxCorner = cNearCenter;
191  for(const CVector3& c_point : {
192  cNearTopLeft, cNearTopRight, cNearBottomLeft, cNearBottomRight,
193  cFarTopLeft, cFarTopRight, cFarBottomLeft, cFarBottomRight
194  }) {
195  if(c_point.GetX() > cBoundingBoxMaxCorner.GetX()) {
196  cBoundingBoxMaxCorner.SetX(c_point.GetX());
197  }
198  if(c_point.GetX() < cBoundingBoxMinCorner.GetX()) {
199  cBoundingBoxMinCorner.SetX(c_point.GetX());
200  }
201  if(c_point.GetY() > cBoundingBoxMaxCorner.GetY()) {
202  cBoundingBoxMaxCorner.SetY(c_point.GetY());
203  }
204  if(c_point.GetY() < cBoundingBoxMinCorner.GetY()) {
205  cBoundingBoxMinCorner.SetY(c_point.GetY());
206  }
207  if(c_point.GetZ() > cBoundingBoxMaxCorner.GetZ()) {
208  cBoundingBoxMaxCorner.SetZ(c_point.GetZ());
209  }
210  if(c_point.GetZ() < cBoundingBoxMinCorner.GetZ()) {
211  cBoundingBoxMinCorner.SetZ(c_point.GetZ());
212  }
213  }
214  cBoundingBoxMaxCorner *= 0.5;
215  cBoundingBoxMinCorner *= 0.5;
216  cBoundingBoxPosition = (cBoundingBoxMaxCorner + cBoundingBoxMinCorner);
217  cBoundingBoxHalfExtents = (cBoundingBoxMaxCorner - cBoundingBoxMinCorner);
218  /* generate frustum planes */
219  arrFrustumPlanes[0].SetFromThreePoints(cNearTopRight, cNearTopLeft, cFarTopLeft);
220  arrFrustumPlanes[1].SetFromThreePoints(cNearBottomLeft, cNearBottomRight, cFarBottomRight);
221  arrFrustumPlanes[2].SetFromThreePoints(cNearTopLeft, cNearBottomLeft, cFarBottomLeft);
222  arrFrustumPlanes[3].SetFromThreePoints(cNearBottomRight, cNearTopRight, cFarBottomRight);
223  arrFrustumPlanes[4].SetFromThreePoints(cNearTopLeft, cNearTopRight, cNearBottomRight);
224  arrFrustumPlanes[5].SetFromThreePoints(cFarTopRight, cFarTopLeft, cFarBottomLeft);
225  /* execute each algorithm */
226  for(CCameraSensorSimulatedAlgorithm* pc_algorithm : s_sensor.Algorithms) {
227  pc_algorithm->Update(s_sensor.ProjectionMatrix,
228  arrFrustumPlanes,
229  cCameraToWorldTransform,
230  cCameraLocation,
231  cBoundingBoxPosition,
232  cBoundingBoxHalfExtents);
233  /* transfer any rays to the controllable entity for rendering */
234  vecCheckedRays.insert(std::end(vecCheckedRays),
235  std::begin(pc_algorithm->GetCheckedRays()),
236  std::end(pc_algorithm->GetCheckedRays()));
237  }
238  }
239  }
