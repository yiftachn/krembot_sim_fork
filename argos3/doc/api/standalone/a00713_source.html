57  void CPositioningDefaultSensor::Update() {
58  m_sReading.Position = m_pcEmbodiedEntity->GetOriginAnchor().Position;
59  if(m_bAddNoise) {
60  m_sReading.Position += CVector3(m_pcRNG->Uniform(m_cPosNoiseRange),
61  m_pcRNG->Uniform(m_cPosNoiseRange),
62  m_pcRNG->Uniform(m_cPosNoiseRange));
63  m_pcEmbodiedEntity->GetOriginAnchor().Orientation.ToAngleAxis(m_cAngle, m_cAxis);
64  m_cAngle += CRadians(m_pcRNG->Uniform(m_cAngleNoiseRange));
65  m_cAxis += CVector3(m_pcRNG->Uniform(m_cAxisNoiseRange),
66  m_pcRNG->Uniform(m_cAxisNoiseRange),
67  m_pcRNG->Uniform(m_cAxisNoiseRange));
68  m_sReading.Orientation.FromAngleAxis(m_cAngle, m_cAxis);
69  }
70  else {
71  m_sReading.Orientation = m_pcEmbodiedEntity->GetOriginAnchor().Orientation;
