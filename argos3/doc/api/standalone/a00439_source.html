321  bool CEmbodiedEntity::MoveTo(const CVector3& c_position,
322  const CQuaternion& c_orientation,
323  bool b_check_only) {
324  /* Can't move an entity with no model associated */
325  if(GetPhysicsModelsNum() == 0) return false;
326  /* Save current position and orientation */
327  CVector3 cOriginalPosition = m_psOriginAnchor->Position;
328  CQuaternion cOriginalOrientation = m_psOriginAnchor->Orientation;
329  /* Treat specially the case of movable entity */
330  if(m_bMovable) {
331  /* Move entity and check for collisions */
332  m_tPhysicsModelVector[0]->MoveTo(c_position, c_orientation);
333  bool bNoCollision = ! m_tPhysicsModelVector[0]->IsCollidingWithSomething();
334  /* Depending on the presence of collisions... */
335  if(bNoCollision && !b_check_only) {
336  /* No collision and not a simple check */
337  /* Tell the caller that we managed to move the entity */
338  return true;
339  }
340  else {
341  /* Collision or just a check, undo changes */
342  m_tPhysicsModelVector[0]->MoveTo(cOriginalPosition, cOriginalOrientation);
343  /* Tell the caller about collisions */
344  return bNoCollision;
345  }
346  }
347  else {
348  /* The entity is not movable, go through all the models */
349  size_t i;
350  bool bNoCollision = true;
351  for(i = 0; i < m_tPhysicsModelVector.size() && bNoCollision; ++i) {
352  m_tPhysicsModelVector[i]->MoveTo(c_position, c_orientation);
353  bNoCollision = !m_tPhysicsModelVector[i]->IsCollidingWithSomething();
354  }
355  if(bNoCollision && !b_check_only) {
356  /* No collision and not a simple check */
357  CalculateBoundingBox();
358  /* Tell the caller that we managed to move the entity */
359  return true;
360  }
361  else {
362  /* No collision or just a check, undo changes */
363  for(size_t j = 0; j < i; ++j) {
364  m_tPhysicsModelVector[j]->MoveTo(cOriginalPosition, cOriginalOrientation);
365  }
366  /* Tell the caller about collisions */
367  return bNoCollision;
368  }
369  }
370  }

380  void CEmbodiedEntity::CalculateBoundingBox() {
381  if(GetPhysicsModelsNum() > 0) {
382  /*
383  * There is at least one physics engine entity associated
384  */
385  if(m_bMovable) {
386  /* The bounding box points directly to the associated model bounding box */
387  m_sBoundingBox = &m_tPhysicsModelVector[0]->GetBoundingBox();
388  }
389  else {
390  /* The bounding box is obtained taking the extrema of all the bboxes of all the engines */
391  if(m_sBoundingBox == NULL) {
392  m_sBoundingBox = new SBoundingBox();
393  }
394  *m_sBoundingBox = m_tPhysicsModelVector[0]->GetBoundingBox();
395  for(size_t i = 1; i < GetPhysicsModelsNum(); ++i) {
396  const SBoundingBox& sBBox = m_tPhysicsModelVector[0]->GetBoundingBox();
397  CHECK_CORNER(Min, X, >);
398  CHECK_CORNER(Min, Y, >);
399  CHECK_CORNER(Min, Z, >);
400  CHECK_CORNER(Max, X, <);
401  CHECK_CORNER(Max, Y, <);
402  CHECK_CORNER(Max, Z, <);
403  }
404  }
405  }
406  else {
407  /*
408  * No physics engine entity associated
409  */
410  if(! m_bMovable && m_sBoundingBox != NULL) {
411  /* A non-movable entity has its own bounding box, delete it */
412  delete m_sBoundingBox;
413  }
414  m_sBoundingBox = NULL;
415  }

421  bool CEmbodiedEntity::IsCollidingWithSomething() const {
422  /* If no model is associated, you can't call this function */

441  return false;
