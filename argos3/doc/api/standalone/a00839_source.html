34  void CDynamics2DEngine::Init(TConfigurationNode& t_tree) {
35  try {
36  /* Init parent */
37  CPhysicsEngine::Init(t_tree);
38  /* Parse XML */
39  GetNodeAttributeOrDefault(t_tree, "elevation", m_fElevation, m_fElevation);
40  if(NodeExists(t_tree, "friction")) {
41  TConfigurationNode& tNode = GetNode(t_tree, "friction");
42  GetNodeAttributeOrDefault(tNode, "box_linear_friction", m_fBoxLinearFriction, m_fBoxLinearFriction);
43  GetNodeAttributeOrDefault(tNode, "box_angular_friction", m_fBoxAngularFriction, m_fBoxAngularFriction);
44  GetNodeAttributeOrDefault(tNode, "cylinder_linear_friction", m_fCylinderLinearFriction, m_fCylinderLinearFriction);
45  GetNodeAttributeOrDefault(tNode, "cylinder_angular_friction", m_fCylinderAngularFriction, m_fCylinderAngularFriction);
46  }
47  /* Override volume top and bottom with the value of m_fElevation */
48  if(!GetVolume().TopFace) GetVolume().TopFace = new SHorizontalFace;
49  if(!GetVolume().BottomFace) GetVolume().BottomFace = new SHorizontalFace;
50  GetVolume().TopFace->Height = m_fElevation;
51  GetVolume().BottomFace->Height = m_fElevation;
52  /* Initialize physics */
53  cpInitChipmunk();
54  cpResetShapeIdCounter();
55  /* Used to attach static geometries so that they won't move and to simulate friction */
56  m_ptGroundBody = cpBodyNew(INFINITY, INFINITY);
57  /* Create the space to contain the movable objects */
58  m_ptSpace = cpSpaceNew();
59  /* Subiterations to solve constraints.
60  The more, the better for precision but the worse for speed
61  */
62  m_ptSpace->iterations = GetIterations();
63  /* Spatial hash */
64  if(NodeExists(t_tree, "spatial_hash")) {
65  TConfigurationNode& tNode = GetNode(t_tree, "spatial_hash");
66  cpFloat fSize;
67  UInt32 unNum;
68  GetNodeAttribute(tNode, "cell_size", fSize);
69  GetNodeAttribute(tNode, "cell_num", unNum);
70  cpSpaceUseSpatialHash(m_ptSpace, fSize, unNum);
71  }
72  /* Gripper-Gripped callback functions */
73  cpSpaceAddCollisionHandler(
74  m_ptSpace,
75  SHAPE_GRIPPER,
76  SHAPE_GRIPPABLE,
77  BeginCollisionBetweenGripperAndGrippable,
78  ManageCollisionBetweenGripperAndGrippable,
79  NULL,
80  NULL,
81  NULL);
82  }
83  catch(CARGoSException& ex) {
84  THROW_ARGOSEXCEPTION_NESTED("Error initializing the dynamics 2D engine \"" << GetId() << "\"", ex);
85  }
86  }

102  void CDynamics2DEngine::Update() {
103  /* Update the physics state from the entities */
104  for(CDynamics2DModel::TMap::iterator it = m_tPhysicsModels.begin();
105  it != m_tPhysicsModels.end(); ++it) {
106  it->second->UpdateFromEntityStatus();
107  }
108  /* Perform the step */
109  for(size_t i = 0; i < GetIterations(); ++i) {
110  for(CDynamics2DModel::TMap::iterator it = m_tPhysicsModels.begin();
111  it != m_tPhysicsModels.end(); ++it) {
112  it->second->UpdatePhysics();
113  }
114  cpSpaceStep(m_ptSpace, GetPhysicsClockTick());
115  }
116  /* Update the simulated space */
117  for(CDynamics2DModel::TMap::iterator it = m_tPhysicsModels.begin();
118  it != m_tPhysicsModels.end(); ++it) {
119  it->second->UpdateEntityStatus();
120  }
121  }

179  static void Dynamics2DSegmentQueryFunc(cpShape* pt_shape, cpFloat f_t, cpVect, void* pt_data) {
180  /* Get the data associated to this query */
181  SDynamics2DSegmentHitData& sData = *reinterpret_cast<SDynamics2DSegmentHitData*>(pt_data);
182  /* Hit found, is it within the limits? */
183  CDynamics2DModel& cModel = *reinterpret_cast<CDynamics2DModel*>(pt_shape->body->data);
184  CVector3 cIntersectionPoint;
185  sData.Ray.GetPoint(cIntersectionPoint, f_t);
186  if((cIntersectionPoint.GetZ() >= cModel.GetBoundingBox().MinCorner.GetZ()) &&
187  (cIntersectionPoint.GetZ() <= cModel.GetBoundingBox().MaxCorner.GetZ()) ) {
188  /* Yes, a real hit */
189  sData.Intersections.push_back(
190  SEmbodiedEntityIntersectionItem(
191  &cModel.GetEmbodiedEntity(),
192  f_t));
193  }
