107  SetLEDMaterial(cColor.GetRed(),
108  cColor.GetGreen(),
109  cColor.GetBlue());
110  glCallList(m_unLEDList);

112  glPopMatrix();

202  cNormal.RotateY(cAngle);
203  }
204  glEnd();
205  /* Tire */
206  cNormal.Set(1.0f, 0.0f, 0.0f);
207  cVertex.Set(WHEEL_RADIUS, 0.0f);
