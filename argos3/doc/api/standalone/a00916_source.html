207  CRay3 CQTOpenGLWidget::RayFromWindowCoord(int n_x,
208  int n_y) {
209  /* Make sure OpenGL context is correct */
210  makeCurrent();
211  /* Rescale coordinates by devicePixelRatio */
212  n_x *= devicePixelRatio();
213  n_y *= devicePixelRatio();
214  /* Get current viewport */
215  GLint nViewport[4];
216  glGetIntegerv(GL_VIEWPORT, nViewport);
217  /* Get OpenGL matrices */
218  GLdouble fModelViewMatrix[16];
219  GLdouble fProjectionMatrix[16];
220  glGetDoublev(GL_MODELVIEW_MATRIX, fModelViewMatrix);
221  glGetDoublev(GL_PROJECTION_MATRIX, fProjectionMatrix);
222  /*
223  * Convert mouse position in window into OpenGL representation
224  */
225  /* The x coordinate stays the same */
226  GLfloat fWinX = n_x;
227  /* The y coordinate of the window is top-left; in OpenGL is bottom-left */
228  GLfloat fWinY = nViewport[3] - n_y;
229  /*
230  * Get the position of the ray start in the world
231  * The ray starts at the near clipping plane (depth = 0.0f)
232  */
233  GLdouble fRayStartX, fRayStartY, fRayStartZ;
234  gluUnProject(fWinX, fWinY, 0.0f,
235  fModelViewMatrix, fProjectionMatrix, nViewport,
236  &fRayStartX, &fRayStartY, &fRayStartZ);
237  /*
238  * Get the position of the ray end in the world
239  * The ray starts at the far clipping plane (depth = 1.0f)
240  */
241  GLdouble fRayEndX, fRayEndY, fRayEndZ;
242  gluUnProject(fWinX, fWinY, 1.0f,
243  fModelViewMatrix, fProjectionMatrix, nViewport,
244  &fRayEndX, &fRayEndY, &fRayEndZ);
245  doneCurrent();
246  return CRay3(CVector3(fRayStartX, fRayStartY, fRayStartZ),
247  CVector3(fRayEndX, fRayEndY, fRayEndZ));
248  }

253  CVector3 CQTOpenGLWidget::GetWindowCoordInWorld(int n_x,
254  int n_y) {
255  /* Make sure OpenGL context is correct */
256  makeCurrent();
257  /* Rescale coordinates by devicePixelRatio */
258  n_x *= devicePixelRatio();
259  n_y *= devicePixelRatio();
260  /* Get current viewport */
261  GLint nViewport[4];
262  glGetIntegerv(GL_VIEWPORT, nViewport);
263  /* Get OpenGL matrices */
264  GLdouble fModelViewMatrix[16];
265  GLdouble fProjectionMatrix[16];
266  glGetDoublev(GL_MODELVIEW_MATRIX, fModelViewMatrix);
267  glGetDoublev(GL_PROJECTION_MATRIX, fProjectionMatrix);
268  /*
269  * Convert mouse position in window into a 3D representation
270  */
271  /* The x coordinate stays the same */
272  GLfloat fWinX = n_x;
273  /* The y coordinate of the window is top-left; in OpenGL is bottom-left */
274  GLfloat fWinY = nViewport[3] - n_y;
275  /* Read the z coordinate from the depth buffer in the back buffer */
276  GLfloat fWinZ;
277  glReadBuffer(GL_BACK);
278  glReadPixels(n_x, (GLint)fWinY, 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &fWinZ);
279  /* Get the actual position in the world */
280  GLdouble fWorldX, fWorldY, fWorldZ;
281  gluUnProject(fWinX, fWinY, fWinZ,
282  fModelViewMatrix, fProjectionMatrix, nViewport,
283  &fWorldX, &fWorldY, &fWorldZ);
284  /*
285  * Swap coordinates when creating the ray
286  * In ARGoS, the up vector is the Z-axis, while in OpenGL it is the Y-axis
287  */
288  doneCurrent();
289  return CVector3(fWorldX, fWorldZ, fWorldY);
290  }

813  void CQTOpenGLWidget::mouseReleaseEvent(QMouseEvent* pc_event) {
814  /*
815  * Mouse grabbed, selected entity, CTRL pressed
816  */
817  if(m_bMouseGrabbed &&
818  m_sSelectionInfo.IsSelected &&
819  (pc_event->modifiers() & Qt::ControlModifier)) {
820  /* Treat selected entity as an embodied entity */
821  CEmbodiedEntity* pcEntity = dynamic_cast<CEmbodiedEntity*>(m_sSelectionInfo.Entity);
822  if(pcEntity == NULL) {
823  /* Treat selected entity as a composable entity with an embodied component */
824  CComposableEntity* pcCompEntity = dynamic_cast<CComposableEntity*>(m_sSelectionInfo.Entity);
825  if(pcCompEntity != NULL && pcCompEntity->HasComponent("body")) {
826  pcEntity = &pcCompEntity->GetComponent<CEmbodiedEntity>("body");
827  }
828  else {
829  /* All conversions failed, get out */
830  m_bMouseGrabbed = false;
831  return;
832  }
833  }
834  /*
835  * If we get here, pcEntity is set to a non-NULL value
836  * Move the entity to the wanted place
837  */
838  /* Create a plane coincident with the world XY plane, centered at the entity position */
839  CPlane cXYPlane(pcEntity->GetOriginAnchor().Position, CVector3::Z);
840  /* Create a ray from the image pixel to the world */
841  CRay3 cMouseRay =
842  RayFromWindowCoord(pc_event->pos().x(),
843  pc_event->pos().y());
844  /* Calculate the new entity position as the intersection of the projected mouse position
845  with the plane created before */
846  CVector3 cNewPos;
847  if(cMouseRay.Intersects(cXYPlane, cNewPos)) {
848  pcEntity->MoveTo(cNewPos,
849  pcEntity->GetOriginAnchor().Orientation);
850  /* Entity moved, redraw */
851  update();
852  }
853  }
854  else {
855  m_cUserFunctions.MouseKeyReleased(pc_event);
856  }
857  /*
858  * Mouse was grabbed, button released -> ungrab mouse
859  */
860  m_bMouseGrabbed = false;
861  }

868  * Moving while mouse grabbed -> camera movement
869  */
