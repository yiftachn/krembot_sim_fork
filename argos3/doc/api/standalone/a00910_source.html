364  GLuint CQTOpenGLObjModel::AddVertex(SInt32 n_key,
365  const CVector3& c_position,
366  const CVector3& c_normal,
367  const CVector2& c_texture) {
368  /* for readability */
369  typedef std::multimap<SInt32, GLuint>::iterator TCacheIterator;
370  typedef std::multimap<SInt32, GLuint>::value_type TCacheEntry;
371  /* find cache entries matching the key */
372  std::pair<TCacheIterator, TCacheIterator> cRange =
373  m_mapVertexCache.equal_range(n_key);
374  /* check if the vertex already exists in the cache */
375  TCacheIterator itVertex =
376  std::find_if(cRange.first,
377  cRange.second,
378  [this, &c_position, &c_normal, &c_texture] (const TCacheEntry& c_entry) {
379  return (m_vecVertices[c_entry.second].Position == c_position &&
380  m_vecVertices[c_entry.second].Normal == c_normal &&
381  m_vecVertices[c_entry.second].Texture == c_texture);
382  });
383  /* if the vertex doesn't exist create it and put it in the cache */
384  if(itVertex == cRange.second) {
385  m_vecVertices.emplace_back(c_position, c_normal, c_texture);
386  itVertex = m_mapVertexCache.emplace(n_key, m_vecVertices.size() - 1);
387  }
388  return itVertex->second;
389  }

394  void CQTOpenGLObjModel::BuildMeshes() {
395  SMaterial::TMapIterator itCurrentMaterial = std::end(m_mapMaterials);
396  for(const STriangle& s_triangle : m_vecTriangles) {
397  if(s_triangle.Material != itCurrentMaterial) {
398  /* select the current material */
399  itCurrentMaterial = s_triangle.Material;
400  /* create a new mesh */
401  m_vecMeshes.emplace_back(itCurrentMaterial);
402  }
403  m_vecMeshes.back().Triangles.push_back(s_triangle.Indices[0]);
404  m_vecMeshes.back().Triangles.push_back(s_triangle.Indices[1]);
405  m_vecMeshes.back().Triangles.push_back(s_triangle.Indices[2]);
406  }
407  /* sort the meshes based on their material's alpha. Fully opaque meshes towards the
408  front and fully transparent towards the back. */
409  std::sort(m_vecMeshes.begin(),
410  m_vecMeshes.end(),
411  [] (const SMesh& s_left,
412  const SMesh& s_right) {
413  return (s_left.Material->second.Alpha > s_right.Material->second.Alpha);
414  });
415  }
