26  void CDifferentialSteeringDefaultSensor::SetRobot(CComposableEntity& c_entity) {

28  m_pcWheeledEntity = &(c_entity.GetComponent<CWheeledEntity>("wheels"));

30  THROW_ARGOSEXCEPTION("The differential steering sensor can be associated only to a robot with 2 wheels");

65  void CDifferentialSteeringDefaultSensor::Update() {
66  m_sReading.VelocityLeftWheel = m_pfWheelVelocities[0] * 100.0f;
67  m_sReading.VelocityRightWheel = m_pfWheelVelocities[1] * 100.0f;
68  m_sReading.CoveredDistanceLeftWheel = m_sReading.VelocityLeftWheel * CPhysicsEngine::GetSimulationClockTick();
69  m_sReading.CoveredDistanceRightWheel = m_sReading.VelocityRightWheel * CPhysicsEngine::GetSimulationClockTick();
70  if(m_bAddNoise) {
71  m_sReading.VelocityLeftWheel += m_pcRNG->Uniform(m_cVelNoiseRange);
72  m_sReading.VelocityRightWheel += m_pcRNG->Uniform(m_cVelNoiseRange);
73  m_sReading.CoveredDistanceLeftWheel += m_pcRNG->Uniform(m_cDistNoiseRange);
74  m_sReading.CoveredDistanceRightWheel += m_pcRNG->Uniform(m_cDistNoiseRange);
75  }
76  }
