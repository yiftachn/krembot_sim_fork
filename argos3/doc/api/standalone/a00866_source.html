80  void CDynamics3DMultiBodyObjectModel::CalculateBoundingBox() {
81  btVector3 cModelAabbMin, cModelAabbMax, cBodyAabbMin, cBodyAabbMax;
82  /* Initialize the bounding box with the base's AABB */
83  std::shared_ptr<CAbstractBody>& ptr_base = m_vecBodies[0];
84  ptr_base->GetShape().getAabb(ptr_base->GetTransform(), cModelAabbMin, cModelAabbMax);
85  /* Extend AABB to include other bodies */
86  for(const std::shared_ptr<CAbstractBody>& ptr_body : m_vecBodies) {
87  /* Get the axis aligned bounding box for the current body */
88  ptr_body->GetShape().getAabb(ptr_body->GetTransform(), cBodyAabbMin, cBodyAabbMax);
89  /* Update minimum corner */
90  cModelAabbMin.setX(cModelAabbMin.getX() < cBodyAabbMin.getX() ? cModelAabbMin.getX() : cBodyAabbMin.getX());
91  cModelAabbMin.setY(cModelAabbMin.getY() < cBodyAabbMin.getY() ? cModelAabbMin.getY() : cBodyAabbMin.getY());
92  cModelAabbMin.setZ(cModelAabbMin.getZ() < cBodyAabbMin.getZ() ? cModelAabbMin.getZ() : cBodyAabbMin.getZ());
93  /* Update maximum corner */
94  cModelAabbMax.setX(cModelAabbMax.getX() > cBodyAabbMax.getX() ? cModelAabbMax.getX() : cBodyAabbMax.getX());
95  cModelAabbMax.setY(cModelAabbMax.getY() > cBodyAabbMax.getY() ? cModelAabbMax.getY() : cBodyAabbMax.getY());
96  cModelAabbMax.setZ(cModelAabbMax.getZ() > cBodyAabbMax.getZ() ? cModelAabbMax.getZ() : cBodyAabbMax.getZ());
97  }
98  /* Write back the bounding box swapping the coordinate systems and the Y component */
99  GetBoundingBox().MinCorner.Set(cModelAabbMin.getX(), -cModelAabbMax.getZ(), cModelAabbMin.getY());
100  GetBoundingBox().MaxCorner.Set(cModelAabbMax.getX(), -cModelAabbMin.getZ(), cModelAabbMax.getY());
101  }

106  void CDynamics3DMultiBodyObjectModel::MoveTo(const CVector3& c_position,
107  const CQuaternion& c_orientation) {
108  /* Calculate the destination transform */
109  const btTransform& cMoveToTransform =
110  btTransform(btQuaternion(c_orientation.GetX(),
111  c_orientation.GetZ(),
112  -c_orientation.GetY(),
113  c_orientation.GetW()),
114  btVector3(c_position.GetX(),
115  c_position.GetZ(),
116  -c_position.GetY()));
117  /* Calculate the origin anchor transform */
118  const SAnchor& sOriginAnchor = GetEmbodiedEntity().GetOriginAnchor();
119  const btTransform& cOriginAnchorTransform =
120  btTransform(btQuaternion(sOriginAnchor.Orientation.GetX(),
121  sOriginAnchor.Orientation.GetZ(),
122  -sOriginAnchor.Orientation.GetY(),
123  sOriginAnchor.Orientation.GetW()),
124  btVector3(sOriginAnchor.Position.GetX(),
125  sOriginAnchor.Position.GetZ(),
126  -sOriginAnchor.Position.GetY()));
127  /* Move the base of the model to the destination */
128  m_cMultiBody.setBaseWorldTransform(cMoveToTransform *
129  cOriginAnchorTransform.inverse() * m_cMultiBody.getBaseWorldTransform());
130  /* Move the remaining bodies */
131  btAlignedObjectArray<btQuaternion> vecRotations;
132  btAlignedObjectArray<btVector3> vecTranslations;
133  m_cMultiBody.updateCollisionObjectWorldTransforms(vecRotations, vecTranslations);
134  /* Update entity state */
135  UpdateEntityStatus();
136  }

141  void CDynamics3DMultiBodyObjectModel::UpdateOriginAnchor(SAnchor& s_anchor) {
142  /* Calculate the origin anchor's position and orientation from the base link's anchor */
143  const SAnchor& sBaseAnchor = m_vecBodies[0]->GetAnchor();
144  /* Calculate orientation */
145  CQuaternion cOriginAnchorOrientation = sBaseAnchor.OffsetOrientation.Inverse() *
146  sBaseAnchor.Orientation;
147  /* Calculate position */
148  CVector3 cOriginAnchorPosition = -sBaseAnchor.OffsetPosition;
149  cOriginAnchorPosition.Rotate(cOriginAnchorOrientation);
150  cOriginAnchorPosition += sBaseAnchor.Position;
151  /* Transfer to origin anchor */
152  s_anchor.Position = cOriginAnchorPosition;
153  s_anchor.Orientation = cOriginAnchorOrientation;
154  }
