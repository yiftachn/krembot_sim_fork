57  void CDynamics3DMagnetismPlugin::Update() {
58  if(m_vecDipoles.size() < 2) {
59  /* Nothing to do */
60  return;
61  }
62  for(std::vector<SMagneticDipole>::iterator itDipole0 = std::begin(m_vecDipoles);
63  itDipole0 != (std::end(m_vecDipoles) - 1);
64  ++itDipole0) {
65  for(std::vector<SMagneticDipole>::iterator itDipole1 = std::next(itDipole0, 1);
66  itDipole1 != std::end(m_vecDipoles);
67  ++itDipole1) {
68  const btTransform& cTransformDipole0 = itDipole0->Offset * itDipole0->Body->GetTransform();
69  const btTransform& cTransformDipole1 = itDipole1->Offset * itDipole1->Body->GetTransform();
70  const btVector3& cPositionDipole0 = cTransformDipole0.getOrigin();
71  const btVector3& cPositionDipole1 = cTransformDipole1.getOrigin();
72  /* calculate the distance between the two magnetic bodies */
73  btScalar fDistance = cPositionDipole0.distance(cPositionDipole1);
74  /* optimization - don't calculate magnetism for dipoles more than m_fMaxDistance apart */
75  if(fDistance > m_fMaxDistance) {
76  continue;
77  }
78  /* perform Barnes-hut algorithm */
79  /* calculate the normalized seperation between the two dipoles, pointing from Dipole1 to Dipole0*/
80  const btVector3& cNormalizedSeparation =
81  btVector3(cPositionDipole0 - cPositionDipole1) / fDistance;
82  /* calculate the rotated field of dipole 0 */
83  const btVector3& cRotatedFieldDipole0 =
84  itDipole0->Body->GetTransform().getBasis() * itDipole0->GetField();
85  /* calculate the rotated field of dipole 1 */
86  const btVector3& cRotatedFieldDipole1 =
87  itDipole1->Body->GetTransform().getBasis() * itDipole1->GetField();
88  /* We now have cRotatedFieldDipole0 and cRotatedFieldDipole1 as the magnetic moments
89  (i.e., m0, m1), cNormalizedSeparation as the direction unit vector from Dipole 1
90  to Dipole 0 (i.e., n), fDistance is the scalar distance between the dipoles (i.e.,
91  d), and B0 is the magnetic flux density at Dipole 0.
92  B0 = u0/4pi * [3n(n.m1) - m1] / d^3
93  T0 = m0 * B0
94  = u0.4pi/d^3 * [3 (m1.n)(m0 * n) - m0 * m1]
95 
96  F0 = grad(m0.B0)
97  = u0.4pi/d^4 * [-15n(m0.n)(m1.n) + 3n(m0.m1) + 3(m0(m1.n)+m1(m0.n))]
98  */
99  /* calculate the intermediate cross and dot products */
100  const btVector3& cCrossProduct01 = cRotatedFieldDipole0.cross(cRotatedFieldDipole1);
101  const btVector3& cCrossProduct0 = cRotatedFieldDipole0.cross(cNormalizedSeparation);
102  const btVector3& cCrossProduct1 = cRotatedFieldDipole1.cross(cNormalizedSeparation);
103  btScalar fDotProduct01 = cRotatedFieldDipole0.dot(cRotatedFieldDipole1);
104  btScalar fDotProduct0 = cRotatedFieldDipole0.dot(cNormalizedSeparation);
105  btScalar fDotProduct1 = cRotatedFieldDipole1.dot(cNormalizedSeparation);
106  /* calculate the magnetic force and torque */
107  const btVector3& cTorque0 =
108  ((3 * fDotProduct1 * cCrossProduct0) - cCrossProduct01) *
109  m_fForceConstant / btPow(fDistance, 3);
110  const btVector3& cTorque1 =
111  ((3 * fDotProduct0 * cCrossProduct1) + cCrossProduct01) *
112  m_fForceConstant / btPow(fDistance, 3);
113  const btVector3& cForce = (m_fForceConstant / btPow(fDistance, 4)) *
114  ((-15 * cNormalizedSeparation * fDotProduct1 * fDotProduct0) +
115  (3 * cNormalizedSeparation * fDotProduct01) +
116  (3 * (fDotProduct1 * cRotatedFieldDipole0 + fDotProduct0 * cRotatedFieldDipole1)));
117  /* apply torques and forces to the bodies */
118  itDipole0->Body->ApplyForce(cForce, (itDipole0->Offset).getOrigin());
119  itDipole0->Body->ApplyTorque(cTorque0);
120  itDipole1->Body->ApplyForce(-cForce, (itDipole1->Offset).getOrigin());
121  itDipole1->Body->ApplyTorque(cTorque1);
122  }
123  }
124  }
