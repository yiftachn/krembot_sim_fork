152  /* Call user init function */
153  if(NodeExists(m_tConfigurationRoot, "loop_functions")) {
154  m_pcLoopFunctions->Init(GetNode(m_tConfigurationRoot, "loop_functions"));
155  }
156  /* Physics engines */
157  InitPhysics2();
158  /* Media */
159  InitMedia2();
160  /* Initialise visualization */
161  TConfigurationNodeIterator itVisualization;

163  ((itVisualization = itVisualization.begin(&GetNode(m_tConfigurationRoot, "visualization"))) != itVisualization.end())) {
164  InitVisualization(GetNode(m_tConfigurationRoot, "visualization"));
165  }
166  else {
167  LOG << "[INFO] No visualization selected." << std::endl;
168  m_pcVisualization = new CDefaultVisualization();
169  }
170  /* Start profiling, if needed */
171  if(IsProfiling()) {
172  m_pcProfiler->Start();
173  }

179  void CSimulator::Reset() {
180  /* Reset terminated flag */
181  m_bTerminated = false;
182  /* if random seed is 0 or is not specified, init with the current timeval */
183  if(m_bWasRandomSeedSet) {
184  CRandom::SetSeedOf("argos", m_unRandomSeed);
185  }
186  else {
187  /* Prepare the default value based on the current clock time */
188  struct timeval sTimeValue;
189  ::gettimeofday(&sTimeValue, NULL);
190  UInt32 unSeed = static_cast<UInt32>(sTimeValue.tv_usec);
191  CRandom::SetSeedOf("argos", unSeed);
192  m_unRandomSeed = unSeed;
193  LOG << "[INFO] Using random seed = " << m_unRandomSeed << std::endl;
194  }
195  CRandom::GetCategory("argos").ResetRNGs();
196  /* Reset the space */
197  m_pcSpace->Reset();
198  /* Reset the media */
199  for(CMedium::TMap::iterator it = m_mapMedia.begin();
200  it != m_mapMedia.end(); ++it) {
201  it->second->Reset();
202  }
203  /* Reset the physics engines */
204  for(CPhysicsEngine::TMap::iterator it = m_mapPhysicsEngines.begin();
205  it != m_mapPhysicsEngines.end(); ++it) {
206  it->second->Reset();
207  }
208  /* Reset the loop functions */
209  m_pcLoopFunctions->Reset();
210  LOG.Flush();
211  LOGERR.Flush();
212  }

309  void CSimulator::InitFramework(TConfigurationNode& t_tree) {
310  try {
311  /* Parse the 'system' node */
312  if(NodeExists(t_tree, "system")) {
313  TConfigurationNode tSystem;
314  tSystem = GetNode(t_tree, "system");
315  GetNodeAttributeOrDefault(tSystem, "threads", m_unThreads, m_unThreads);
316  if(m_unThreads == 0) {
317  LOG << "[INFO] Not using threads" << std::endl;
318  m_pcSpace = new CSpaceNoThreads();
319  }
320  else {
321  LOG << "[INFO] Using " << m_unThreads << " parallel threads" << std::endl;
322  std::string strThreadingMethod = "balance_quantity";
323  GetNodeAttributeOrDefault(tSystem, "method", strThreadingMethod, strThreadingMethod);
324  if(strThreadingMethod == "balance_quantity") {
325  LOG << "[INFO] Chosen method \"balance_quantity\": threads will be assigned the same"
326  << std::endl
327  << "[INFO] number of tasks, independently of the task length."
328  << std::endl;
329  m_pcSpace = new CSpaceMultiThreadBalanceQuantity();
330  }
331  else if(strThreadingMethod == "balance_length") {
332  LOG << "[INFO] Chosen method \"balance_length\": threads will be assigned different"
333  << std::endl
334  << "[INFO] numbers of tasks, depending on the task length."
335  << std::endl;
336  m_pcSpace = new CSpaceMultiThreadBalanceLength();
337  }
338  else {
339  THROW_ARGOSEXCEPTION("Error parsing the <system> tag. Unknown threading method \"" << strThreadingMethod << "\". Available methods: \"balance_quantity\" and \"balance_length\".");
340  }
341  }
342  }
343  else {
344  LOG << "[INFO] Not using threads" << std::endl;
345  m_pcSpace = new CSpaceNoThreads();
346  }
347  /* Get 'experiment' node */
348  TConfigurationNode tExperiment;
349  tExperiment = GetNode(t_tree, "experiment");
350  /* Parse random seed */
351  /* Buffer to hold the random seed */
352  if(!m_bWasRandomSeedSet)
353  GetNodeAttributeOrDefault(tExperiment,
354  "random_seed",
355  m_unRandomSeed,
356  static_cast<UInt32>(0));
357  /* if random seed is 0 or is not specified, init with the current timeval */
358  if(m_unRandomSeed != 0) {
359  CRandom::CreateCategory("argos", m_unRandomSeed);
360  LOG << "[INFO] Using random seed = " << m_unRandomSeed << std::endl;
361  m_bWasRandomSeedSet = true;
362  }
363  else {
364  /* Prepare the default value based on the current clock time */
365  m_bWasRandomSeedSet = false;
366  struct timeval sTimeValue;
367  ::gettimeofday(&sTimeValue, NULL);
368  UInt32 unSeed = static_cast<UInt32>(sTimeValue.tv_usec);
369  m_unRandomSeed = unSeed;
370  CRandom::CreateCategory("argos", unSeed);
371  LOG << "[INFO] Using random seed = " << unSeed << std::endl;
372  }
373  m_pcRNG = CRandom::CreateRNG("argos");
374  /* Set the simulation clock tick length */
375  GetNodeAttribute(tExperiment,
376  "ticks_per_second",
377  m_unTicksPerSec);
378  CPhysicsEngine::SetSimulationClockTick(1.0 / static_cast<Real>(m_unTicksPerSec));
379  /* Set the maximum simulation duration (in seconds) */
380  Real fExpLength;
381  GetNodeAttributeOrDefault<Real>(tExperiment,
382  "length",
383  fExpLength,
384  0.0f);
385  m_unMaxSimulationClock = fExpLength * m_unTicksPerSec;
386  LOG << "[INFO] Total experiment length in clock ticks = "
387  << (m_unMaxSimulationClock ? ToString(m_unMaxSimulationClock) : "unlimited")
388  << std::endl;
389  /* Check for the 'real_time' attribute */
390  GetNodeAttributeOrDefault(tExperiment, "real_time", m_bRealTimeClock, m_bRealTimeClock);
391  if(m_bRealTimeClock) {
392  LOG << "[INFO] Using the real-time clock." << std::endl;
393  }
394  /* Get the profiling tag, if present */
395  if(NodeExists(t_tree, "profiling")) {
396  TConfigurationNode& tProfiling = GetNode(t_tree, "profiling");
397  std::string strFile;
398  GetNodeAttribute(tProfiling, "file", strFile);
399  std::string strFormat;
400  GetNodeAttribute(tProfiling, "format", strFormat);
401  if(strFormat == "human_readable") {
402  m_bHumanReadableProfile = true;
403  }
404  else if(strFormat == "table") {
405  m_bHumanReadableProfile = false;
406  }
407  else {
408  THROW_ARGOSEXCEPTION("Unrecognized profile format \"" << strFormat << "\". Accepted values are \"human_readable\" and \"table\".");
409  }
410  bool bTrunc = true;
411  GetNodeAttributeOrDefault(tProfiling, "truncate_file", bTrunc, bTrunc);
412  m_pcProfiler = new CProfiler(strFile, bTrunc);
413  }
414  }
415  catch(CARGoSException& ex) {
416  THROW_ARGOSEXCEPTION_NESTED("Failed to initialize the simulator. Parse error inside the <framework> tag.", ex);
417  }
418  }

441  void CSimulator::InitControllers(TConfigurationNode& t_tree) {
442  /*
443  * Go through controllers, loading the library of each of them
444  * and storing type, id and XML tree of each of them for later use
445  */
446  if(! t_tree.NoChildren()) {
447  try {
448  std::string strLibrary;
449  std::string strId;
450  TConfigurationNodeIterator it;
451  for(it = it.begin(&t_tree);
452  it != it.end(); ++it) {
453  /* Get controller id */
454  try {
455  GetNodeAttribute(*it, "id", strId);
456  }
457  catch(CARGoSException& ex) {
458  std::string strValue;
459  it->GetValue(&strValue);
460  THROW_ARGOSEXCEPTION_NESTED("Controller type \"" << strValue << "\" has no assigned id.", ex);
461  }
462  /* Bomb out if id is already in map */
463  if(m_mapControllerConfig.find(strId) != m_mapControllerConfig.end()) {
464  THROW_ARGOSEXCEPTION("Controller id \"" << strId << "\" duplicated");
465  }
466  /* Optionally, process "library" attribute if present */
467  if(NodeAttributeExists(*it, "library")) {
468  /* Get library name */
469  GetNodeAttribute(*it, "library", strLibrary);
470  /* Load library */
471  CDynamicLoading::LoadLibrary(strLibrary);
472  }
473  /* Store XML info in map by id */
474  m_mapControllerConfig.insert(std::pair<std::string, TConfigurationNode*>(strId, &(*it)));
475  }
476  }
477  catch(CARGoSException& ex) {
478  THROW_ARGOSEXCEPTION_NESTED("Error initializing controllers", ex);
479  }
480  }

498  void CSimulator::InitPhysics(TConfigurationNode& t_tree) {
499  try {
500  /* Cycle through the physics engines */
501  TConfigurationNodeIterator itEngines;
502  for(itEngines = itEngines.begin(&t_tree);
503  itEngines != itEngines.end();
504  ++itEngines) {
505  /* Create the physics engine */
506  CPhysicsEngine* pcEngine = CFactory<CPhysicsEngine>::New(itEngines->Value());
507  try {
508  /* Initialize the engine */
509  pcEngine->Init(*itEngines);
510  /* Check that an engine with that ID does not exist yet */
511  if(m_mapPhysicsEngines.find(pcEngine->GetId()) == m_mapPhysicsEngines.end()) {
512  /* Add it to the lists */
513  m_mapPhysicsEngines[pcEngine->GetId()] = pcEngine;
514  m_vecPhysicsEngines.push_back(pcEngine);
515  }
516  else {
517  /* Duplicate id -> error */
518  THROW_ARGOSEXCEPTION("A physics engine with id \"" << pcEngine->GetId() << "\" exists already. The ids must be unique!");
519  }
520  }
521  catch(CARGoSException& ex) {
522  /* Error while executing engine init, destroy what done to prevent memory leaks */
523  pcEngine->Destroy();
524  delete pcEngine;
525  THROW_ARGOSEXCEPTION_NESTED("Error initializing physics engine type \"" << itEngines->Value() << "\"", ex);
526  }
527  }
528  }
529  catch(CARGoSException& ex) {
530  THROW_ARGOSEXCEPTION_NESTED("Failed to initialize the physics engines. Parse error in the <physics_engines> subtree.", ex);
531  }
