72  void CDynamics3DEngine::Reset() {
73  /* Remove and reset all physics models */
74  for(CDynamics3DModel::TMap::iterator itModel = std::begin(m_tPhysicsModels);
75  itModel != std::end(m_tPhysicsModels);
76  ++itModel) {
77  /* Remove model from plugins */
78  for(CDynamics3DPlugin::TMap::iterator itPlugin = std::begin(m_tPhysicsPlugins);
79  itPlugin != std::end(m_tPhysicsPlugins);
80  ++itPlugin) {
81  itPlugin->second->UnregisterModel(*itModel->second);
82  }
83  /* Reset the model */
84  itModel->second->RemoveFromWorld(m_cWorld);
85  itModel->second->Reset();
86  }
87  /* Run the destructors on bullet's components */
88  m_cWorld.~btMultiBodyDynamicsWorld();
89  m_cSolver.~btMultiBodyConstraintSolver();
90  m_cDispatcher.~btCollisionDispatcher();
91  m_cConfiguration.~btDefaultCollisionConfiguration();
92  m_cBroadphase.~btDbvtBroadphase();
93  /* Rerun the constructors for the bullet's components */
94  new (&m_cBroadphase) btDbvtBroadphase;
95  new (&m_cConfiguration) btDefaultCollisionConfiguration;
96  new (&m_cDispatcher) btCollisionDispatcher(&m_cConfiguration);
97  new (&m_cSolver) btMultiBodyConstraintSolver;
98  new (&m_cWorld) btMultiBodyDynamicsWorld(&m_cDispatcher,
99  &m_cBroadphase,
100  &m_cSolver,
101  &m_cConfiguration);
102  /* Provide the same random seed to the solver */
103  m_cSolver.setRandSeed(m_pcRNG->Uniform(m_cRandomSeedRange));
104  /* Reset the plugins */
105  for(CDynamics3DPlugin::TMap::iterator itPlugin = std::begin(m_tPhysicsPlugins);
106  itPlugin != std::end(m_tPhysicsPlugins);
107  ++itPlugin) {
108  itPlugin->second->Reset();
109  }
110  /* Set up the call back for the plugins */
111  m_cWorld.setInternalTickCallback([] (btDynamicsWorld* pc_world, btScalar f_time_step) {
112  CDynamics3DEngine* pc_engine = static_cast<CDynamics3DEngine*>(pc_world->getWorldUserInfo());
113  pc_world->clearForces();
114  for(std::pair<const std::string, CDynamics3DPlugin*>& c_plugin :
115  pc_engine->GetPhysicsPlugins()) {
116  c_plugin.second->Update();
117  }
118  }, static_cast<void*>(this), true);
119  /* Add the models back into the engine */
120  for(CDynamics3DModel::TMap::iterator itModel = std::begin(m_tPhysicsModels);
121  itModel != std::end(m_tPhysicsModels);
122  ++itModel) {
123  /* Add model to plugins */
124  for(CDynamics3DPlugin::TMap::iterator itPlugin = std::begin(m_tPhysicsPlugins);
125  itPlugin != std::end(m_tPhysicsPlugins);
126  ++itPlugin) {
127  itPlugin->second->RegisterModel(*itModel->second);
128  }
129  /* Add model to world */
130  itModel->second->AddToWorld(m_cWorld);
131  }
132  /* Initialize any multi-body constraints */
133  for (SInt32 i = 0; i < m_cWorld.getNumMultiBodyConstraints(); i++) {
134  m_cWorld.getMultiBodyConstraint(i)->finalizeMultiDof();
135  }
136  }

171  void CDynamics3DEngine::Update() {
172  /* Update the physics state from the entities */
173  for(CDynamics3DModel::TMap::iterator it = m_tPhysicsModels.begin();
174  it != std::end(m_tPhysicsModels); ++it) {
175  it->second->UpdateFromEntityStatus();
176  }
177  /* Step the simuation forwards */
178  m_cWorld.stepSimulation(GetSimulationClockTick(),
179  GetIterations(),
180  GetPhysicsClockTick());
181  /* Update the simulated space */
182  for(CDynamics3DModel::TMap::iterator it = std::begin(m_tPhysicsModels);
183  it != std::end(m_tPhysicsModels);
184  ++it) {
185  it->second->UpdateEntityStatus();
186  }
187  /* Dump the state of the world to a bullet file (if requested) */
188  if(!m_strDebugFilename.empty()) {
189  btDefaultSerializer cSerializer;
190  m_cWorld.serialize(&cSerializer);
191  std::ofstream cDebugOutput(m_strDebugFilename);
192  if(cDebugOutput.is_open()) {
193  cDebugOutput.write(reinterpret_cast<const char*>(cSerializer.getBufferPointer()),
194  cSerializer.getCurrentBufferSize());

202  void CDynamics3DEngine::CheckIntersectionWithRay(TEmbodiedEntityIntersectionData& t_data,
203  const CRay3& c_ray) const {
204  /* Convert the start and end ray vectors to the bullet coordinate system */
205  btVector3 cRayStart(c_ray.GetStart().GetX(), c_ray.GetStart().GetZ(), -c_ray.GetStart().GetY());
206  btVector3 cRayEnd(c_ray.GetEnd().GetX(), c_ray.GetEnd().GetZ(), -c_ray.GetEnd().GetY());
207  btCollisionWorld::ClosestRayResultCallback cResult(cRayStart, cRayEnd);
208  /* The default flag/algorithm 'kF_UseSubSimplexConvexCastRaytest' is too approximate for
209  our purposes */
210  cResult.m_flags |= btTriangleRaycastCallback::kF_UseGjkConvexCastRaytest;
211  /* Run the ray test */
212  m_cWorld.rayTest(cRayStart, cRayEnd, cResult);
213  /* Examine the results */
214  if (cResult.hasHit() && cResult.m_collisionObject->getUserPointer() != nullptr) {
215  Real f_t = (cResult.m_hitPointWorld - cRayStart).length() / c_ray.GetLength();
216  CDynamics3DModel* pcModel =
217  static_cast<CDynamics3DModel*>(cResult.m_collisionObject->getUserPointer());
218  t_data.push_back(SEmbodiedEntityIntersectionItem(&(pcModel->GetEmbodiedEntity()), f_t));
219  }
220  }
